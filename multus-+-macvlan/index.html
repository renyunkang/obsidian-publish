    apiVersion: "k8s.cni.cncf.io/v1"
kind: NetworkAttachmentDefinition
metadata:
  name: macvlan-whereabouts
spec:
  config: '{
      "cniVersion": "0.3.0",
      "type": "macvlan",
      "master": "eth2.73",
      "mode": "bridge",
      "ipam": {
        "type": "whereabouts",
        "range": "172.31.73.1/24",
        "routes": [
          { "dst": "172.31.70.0/24", "gw": "172.31.73.1" },
          { "dst": "172.32.0.0/16", "gw": "172.31.73.1" }
        ],
        "exclude": [
           "172.31.73.1/32"
        ]
      }
    }'

    配置说明：  - “type”: “macvlan”, # 插件名称，可以在/opt/cni/bin下查看支持哪些插件  - “master”: “eth2.73”, # 指定网卡，如果是access口，则直接写网卡名称即可，如 eth2，如果是trunk，则需要看设置的vlan id，如 eth2.73，表示的是 vlan 73  - “mode”: “bridge”, # macvlan的通信模式，可以去查阅一下，有五种模式：bridge，vepa，private，source，passthru，常用的为bridge和vepa模式，这里简单说下它俩的区别，bridge网络流量由系统网卡做转发，而vepa则是需要交换做转发，效率更加高效，但同时要求交换机能够支持发夹（hairpin） 功能，如果是bridge模式，pod是无法ping通宿主机网卡的，这点需要了解一下  - ipam 使用的 ipam 插件的配置，不同的 cni 支持的配置是不同的。官方提供的 ipam 有 dhcp、host-local、static 三种。https://www.cni.dev/plugins/current/ipam/  - dhcp 需要额外的 dhcp 服务器分配 ip  - host-local 适用于单节点的集群，将分配记录存在本地数据库  - static 为容器分配静态 ip 地址  - addresses 配置的地址信息  - routes 配置路由信息  - dns 配置 dns 信息

    whereabouts 社区为改进 host-local 的一个适用于多节点集群的 ipam 插件，支持 ipv4 和 ipv6，同时也内置支持 static 插件路由/dns的配置

    • “type”: “whereabouts” # ipam 插件名称，二进制文件存放在/opt/cni/bin
    • “range”：pod cidr 地址
    • “range_start”：ip 地址段起始地址
    • “range_end”：ip 地址段结束地址
    • “gateway”：网关地址
    • “ipRange”：多个地址段
    • “exclude”：预留/屏蔽的 ip 地址

    apiVersion: apps/v1
kind: Deployment
metadata:
  name: netools2
spec:
  replicas: 2
  selector:
    matchLabels:
      app: netools2
  template:
    metadata:
      annotations:
        k8s.v1.cni.cncf.io/networks: macvlan-whereabouts
        # k8s.v1.cni.cncf.io/networks: macvlan-whereabouts@eth1
        # k8s.v1.cni.cncf.io/networks: default/macvlan-whereabouts
        # k8s.v1.cni.cncf.io/networks: default/macvlan-whereabouts@eth1
        # k8s.v1.cni.cncf.io/networks: '[{"name": "macvlan-whereabouts", "interface": "eth1"}]'
        # k8s.v1.cni.cncf.io/networks: '[{"name": "macvlan-whereabouts", "namespace": "default", "interface": "eth1"}]'
        # k8s.v1.cni.cncf.io/networks: macvlan-conf-1, macvlan-conf-2
        # k8s.v1.cni.cncf.io/networks: '[{"name": "macvlan-conf-1"}, {"name": "macvlan-conf-2"}]'
        # k8s.v1.cni.cncf.io/networks: '[{"name": "macvlan-whereabouts", "default-route": ["192.168.2.1"]}]'

      labels:
        app: netools2
    spec:
      containers:
        - image: rykren/netools
          name: netools2
          ports:
            - containerPort: 80
