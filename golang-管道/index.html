    package main

import (
	"fmt"
	"sync"
)

func main() {
	var (
		wg = sync.WaitGroup{}
		wg1 = sync.WaitGroup{}
		num = 0
		ch = make(chan int)
	)

	wg.Add(2)
	wg1.Add(1)
	go func() {
		for i := 0; i < 200000; i++ {
			ch <- 1
		}
		wg.Done()
	}()

	go func() {
		for i := 0; i < 100000; i++ {
			ch <- 2
		}
		wg.Done()
	}()

	go add(&num, ch, &wg1)

	wg.Wait()
	close(ch)
	wg1.Wait()

	fmt.Printf("num = %d\n", num)
}

func add(num *int, ch chan int, wg *sync.WaitGroup) {
	for v := range ch {
		*num = *num + v
	}
	wg.Done()
}
