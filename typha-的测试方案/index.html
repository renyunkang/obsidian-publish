import matplotlib.pyplot as plt

def plot_graph(log_file, label):
    with open(log_file, 'r') as f:
        lines = f.readlines()

    timestamps = []
    values = []

    # Skip the first line (header)
    for line in lines[3:-2]:
        parts = line.split()
        if len(parts) >= 9:  # Ensure that the line has at least 9 columns
            timestamps.append(parts[0])
            values.append(float(parts[7]))  # Adjust the index based on the column you want to plot

    plt.plot(timestamps, values, label=label)
    plt.xlabel('Timestamp')
    plt.ylabel('Value')  # You may want to adjust the label based on the column you're plotting
    plt.title('Process Resource Usage Over Time')
    plt.legend()
    plt.show()

plot_graph('cpu_usage.log', 'CPU Usage (%)')
plot_graph('mem_usage.log', 'Memory Usage (KB)')
