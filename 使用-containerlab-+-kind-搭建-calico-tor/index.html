        node-ip: 10.1.5.11 # 节点 IP
        node-labels: "rack=rack1" # 节点标签

  extraPortMappings:
  - containerPort: 30880
    hostPort: 30880
    listenAddress: "0.0.0.0" # Optional, defaults to "0.0.0.0"
    protocol: tcp # Optional, defaults to tcp
  - containerPort: 30881
    hostPort: 30881
    listenAddress: "0.0.0.0" # Optional, defaults to "0.0.0.0"
    protocol: tcp # Optional, defaults to tcp
- role: worker

  - |
    kind: JoinConfiguration
    nodeRegistration:
      kubeletExtraArgs:
        node-ip: 10.1.5.12 # 节点 IP
        node-labels: "rack=rack1" # 节点标签    
  image: kindest/node:v1.24.7@sha256:577c630ce8e509131eab1aea12c022190978dd2f745aac5eb1fe65c0807eb315

  - |
    kind: JoinConfiguration
    nodeRegistration:
      kubeletExtraArgs:
        node-ip: 10.1.8.13 # 节点 IP
        node-labels: "rack=rack2" # 节点标签    
  image: kindest/node:v1.24.7@sha256:577c630ce8e509131eab1aea12c022190978dd2f745aac5eb1fe65c0807eb315

  kubeadmConfigPatches:
  - |
    kind: JoinConfiguration
    nodeRegistration:
      kubeletExtraArgs:
        node-ip: 10.1.8.14 # 节点 IP

  image: kindest/node:v1.24.7@sha256:577c630ce8e509131eab1aea12c022190978dd2f745aac5eb1fe65c0807eb315

  disableDefaultCNI: true
  podSubnet: 10.233.64.0/18

  links:

# spine1
set interfaces ethernet eth0 address '172.20.20.2/24'
set interfaces ethernet eth0 description public
set protocols static route 0.0.0.0/0 next-hop 172.20.20.1 distance 1
set nat source rule 100 outbound-interface 'eth0'
set nat source rule 100 source address '10.1.0.0/16'
set nat source rule 100 translation address 'masquerade'

set interfaces ethernet eth1 address '10.1.10.2/24'
set interfaces ethernet eth2 address '10.1.34.2/24'
set protocols bgp parameters router-id '10.1.10.2'
set protocols bgp local-as 500 
set protocols bgp neighbor 10.1.10.1 remote-as '65005'
set protocols bgp neighbor 10.1.34.1 remote-as '65008'


# spine2
set interfaces ethernet eth0 address '172.20.20.6/24'
set interfaces ethernet eth0 description public
set protocols static route 0.0.0.0/0 next-hop 172.20.20.1 distance 1
set nat source rule 100 outbound-interface 'eth0'
set nat source rule 100 source address '10.1.0.0/16'
set nat source rule 100 translation address 'masquerade'

set interfaces ethernet eth1 address '10.1.12.2/24'
set interfaces ethernet eth2 address '10.1.11.2/24'
set protocols bgp parameters router-id '10.1.12.2'
set protocols bgp local-as 800 
set protocols bgp neighbor 10.1.11.1 remote-as '65008'
set protocols bgp neighbor 10.1.12.1 remote-as '65005'


# leaf-1
set interfaces ethernet eth0 disable
set interfaces bridge br0 address '10.1.5.1/24'
set interfaces bridge br0 description 'leaf1-br'
set interfaces bridge br0 member interface eth3
set interfaces bridge br0 member interface eth4

set protocols static route 10.1.34.0/24 next-hop 10.1.10.2
set protocols static route 10.1.11.0/24 next-hop 10.1.12.2
set protocols static route 0.0.0.0/0 next-hop 10.1.10.2
set protocols static route 0.0.0.0/0 next-hop 10.1.12.2

set interfaces ethernet eth1 address '10.1.10.1/24'
set interfaces ethernet eth2 address '10.1.12.1/24'
set protocols bgp local-as 65005
set protocols bgp address-family ipv4-unicast network 10.1.5.0/24
set protocols bgp neighbor 10.1.5.11 address-family ipv4-unicast route-reflector-client
set protocols bgp neighbor 10.1.5.11 remote-as '65005'
set protocols bgp neighbor 10.1.5.12 address-family ipv4-unicast route-reflector-client
set protocols bgp neighbor 10.1.5.12 remote-as '65005'
set protocols bgp neighbor 10.1.10.2 remote-as '500'
set protocols bgp neighbor 10.1.12.2 remote-as '800'
set protocols bgp parameters bestpath as-path multipath-relax
set protocols bgp parameters router-id '10.1.5.1'


# leaf2
set interfaces ethernet eth0 disable
set interfaces bridge br0 address '10.1.8.1/24'
set interfaces bridge br0 description 'leaf2-br'
set interfaces bridge br0 member interface eth3
set interfaces bridge br0 member interface eth4

set protocols static route 10.1.10.0/24 next-hop 10.1.34.2
set protocols static route 10.1.12.0/24 next-hop 10.1.11.2
set protocols static route 0.0.0.0/0 next-hop 10.1.34.2
set protocols static route 0.0.0.0/0 next-hop 10.1.11.2

set interfaces ethernet eth1 address '10.1.34.1/24'
set interfaces ethernet eth2 address '10.1.11.1/24'

set protocols bgp address-family ipv4-unicast network 10.1.8.0/24

set protocols bgp neighbor 10.1.8.13 remote-as '65008'
