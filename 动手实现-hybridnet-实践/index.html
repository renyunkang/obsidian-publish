    root@node:~/clab# cat clab.yaml 
name: cluster
mgmt:
  network: bridge
topology:
  nodes:
    leaf:
      kind: linux
      image: rykren/vyos:1.4
      cmd: /sbin/init
      binds:
        - /lib/modules:/lib/modules
        - vyos/config:/opt/vyatta/etc/config

    server1:
      kind: linux
      image: rykren/nettools:latest
      network-mode: none
      exec:
      - ip link set dev eth1 address aa:c1:ab:9f:49:82
      - ip addr add 10.10.0.2/24 dev eth1
      - ip route replace default via 10.10.0.1
      # add vlan20
      - ip link add link eth1 name eth1.20 type vlan id 20
      - ip addr add 10.10.20.10/24 dev eth1.20 noprefixroute
      - ip link set dev eth1.20 up
      # add vxlan
      - ip link add eth1.vxlan4 type vxlan id 4 dstport 4789 group 239.1.1.1 dev eth1 learning
      # - ip link add eth1.vxlan4 type vxlan id 4 dstport 4789 remote 10.10.10.2 local 10.10.0.2 dev eth1 learning
      # - ip link add eth1.vxlan4 type vxlan id 4 dstport 4789 local 10.10.0.2 dev eth1
      - ip link set eth1.vxlan4 address aa:c1:ab:9f:49:82
      - ip addr add 10.10.0.2/24 brd 10.10.10.255 dev eth1.vxlan4 noprefixroute
      - ip link set eth1.vxlan4 up
      # add container 1
      - ip netns add ns1
      - ip link add name veth01 type veth peer name port01
      - ip link set dev veth01 netns ns1
      - ip netns exec ns1 ip addr add 10.233.100.10/24 dev veth01
      - ip netns exec ns1 ip link set dev veth01 address 32:d1:ae:e8:8e:10
      - ip netns exec ns1 ip link set lo up
      - ip netns exec ns1 ip link set veth01 up
      - ip link set port01 up
      - ip netns exec ns1 ip route add 169.254.1.1 dev  veth01 scope link 
      - ip netns exec ns1 ip route add default via 169.254.1.1 dev veth01
      - bash -c "echo 1 > /proc/sys/net/ipv4/conf/port01/proxy_arp"
      - bash -c "echo 1 > /proc/sys/net/ipv4/conf/port01/forwarding"
      - bash -c "echo 1 > /proc/sys/net/ipv4/conf/port01/route_localnet"
      # add container 2
      - ip netns add ns2
      - ip link add name veth02 type veth peer name port02
      - ip link set dev veth02 netns ns2
      - ip netns exec ns2 ip link set veth02 up
      - ip netns exec ns2 ip link set lo up
      - ip netns exec ns2 ip addr add 10.10.20.10/24 dev veth02
      - ip link set port02 up
      - ip netns exec ns2 ip route add 169.254.1.1 dev  veth02 scope link 
      - ip netns exec ns2 ip route add default via 169.254.1.1 dev veth02
      - bash -c "echo 1 > /proc/sys/net/ipv4/conf/port02/proxy_arp"
      - bash -c "echo 1 > /proc/sys/net/ipv4/conf/port02/forwarding"
      - bash -c "echo 1 > /proc/sys/net/ipv4/conf/port02/route_localnet"
      # add router
      - ip rule add priority 1 table 39999
      - ip route add 10.233.100.10 dev port01  table 39999
      - ip route add 10.10.20.10 dev port02  table 39999
      - ip rule add priority 2 table 40000
      - ip route add 10.233.100.0/24 dev eth1.vxlan4 table 40000
      - ip rule add priority 3 fwmark 0x20/0x20 table 40001
      - ip route add default dev eth1.vxlan4 table 40001
      - ip rule add priority 4 from 10.233.100.0/24 fwmark 0x0/0x4040 table 10000
      - ip route add 10.10.20.0/24 dev eth1.vxlan4 table 10000
      - ip route add throw 10.10.20.1 table 10000
      - ip rule add priority 5 from 10.10.20.0/24 fwmark 0x0/0x4040 table 10001
      - ip route add 10.10.20.0/24 dev eth1.20 scope link table 10001
      - ip route add default via 10.10.20.1 dev eth1.20 table 10001
      # add fdb
      - bridge fdb append 00:00:00:00:00:00 dev eth1.vxlan4 dst 10.10.0.2 self permanent
      - bridge fdb append 00:00:00:00:00:00 dev eth1.vxlan4 dst 10.10.10.2 self permanent
      - bridge fdb append 00:00:00:00:00:00 dev eth1.vxlan4 dst 10.10.11.2 self permanent
      - bridge fdb append aa:c1:ab:9f:49:82 dev eth1.vxlan4 dst 10.10.0.2 self permanent
      - bridge fdb append aa:c1:ab:f9:13:8a dev eth1.vxlan4 dst 10.10.10.2 self permanent
      - bridge fdb append aa:c1:ab:d5:62:23 dev eth1.vxlan4 dst 10.10.11.2 self permanent
      # add neigh proxy
      - ip neighbor add proxy 10.233.100.10 lladdr 32:d1:ae:e8:8e:10 dev eth1.vxlan4
