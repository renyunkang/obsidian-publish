iface lo inet loopback

auto eth0
iface eth0 inet dhcp

auto eth3
iface eth3 inet static
address 100.0.0.1
netmask 255.255.255.252

auto eth1
iface eth1 inet manual

auto eth2
iface eth2 inet manual

auto br0
iface br0 inet static
address 172.16.0.1
network 172.16.0.0
netmask 255.255.255.0
broadcast 172.16.0.255
bridge_ports eth1
bridge_ports eth2
bridge_stp off
bridge_fd 0
bridge_maxwait 0
bridge_maxage 12

root@i-1medguzr:~# cat /etc/bird/bird.conf 
# This is a minimal configuration file, which allows the bird daemon to start
# but will not cause anything else to happen.
#
# Please refer to the documentation in the bird-doc package or BIRD User's
# Guide on http://bird.network.cz/ for more information on configuring BIRD and
# adding routing protocols.

# Change this into your BIRD router ID. It's a world-wide unique identification
# of your router, usually one of router's IPv4 addresses.
router id 172.16.10.1;

# The Kernel protocol is not a real routing protocol. Instead of communicating
# with other routers in the network, it performs synchronization of BIRD's
# routing tables with the OS kernel.
protocol kernel {
  learn;             # Learn all alien routes from the kernel
  persist;           # Don't remove routes on bird shutdown
  scan time 2;       # Scan kernel routing table every 2 seconds
  import all;
  export all; # Default is export none
  graceful restart;  # Turn on graceful restart
  merge paths on;    # Allow export multipath routes (ECMP)
}

# The Device protocol is not a real routing protocol. It doesn't generate any
# routes and it only serves as a module for getting information about network
# interfaces from the kernel. 
# Watch interface up/down events.
protocol device {
        scan time 2;
}


template bgp bgp_template {
        debug all;                                  # debug BGP
        description "BIRD RR";
        local as 64512;                             # the AS used by the local BGP speaker
        multihop;                                   # was not default in 1.3.11
        gateway recursive;
        import all;
        export all;
        add paths on;
        graceful restart;
        connect delay time 2;
        connect retry time 5;
        error wait time 5,30;
        source address 172.16.10.1;                # Source BGP from this IP address

        #rr client;                                  # make defined neighbors rr clients
        #rr cluster id 172.16.10.1;
        #igp table master;                           # necessary to resolve the next-hop
        #hold time 30;                               # hold time send in BGP messages
}

# For peer /host/node/peer_v4/172.16.0.1
protocol bgp Node_172_16_0_1 from bgp_template {
  neighbor 172.16.0.1 as 64512;
}

# For peer /host/node/peer_v4/172.16.10.13
protocol bgp Node_172_16_10_13 from bgp_template {
  neighbor 172.16.10.13 as 64512;
  rr client;
  rr cluster id 172.16.10.1;
}


# For peer /host/node/peer_v4/172.16.10.14
protocol bgp Node_172_16_10_14 from bgp_template {
  neighbor 172.16.10.14 as 64512;
  rr client;
  rr cluster id 172.16.10.1;
}
