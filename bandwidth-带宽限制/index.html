      "ingressBurst": 456,
      "egressRate": 123,
      "egressBurst": 456
    }
  ]
}
字段解释
ingressRate流量可以进入 interface 的速率 bps
ingressBurst使令牌立即可用的最大最大比特数（瞬间峰值）
egressRate流量可以离开接口的速率 bps
egressBurst使令牌立即可用的最大比特数（瞬间峰值）

以上rate 的单位为 bps。必须同时设置 ingressRate 和 ingressBurst 以限制入口带宽，同时设置 egressRate 和 egressBurst 以限制出口带宽。

即使给 pod 设置了kubernetes.io/ingress-bandwidth and kubernetes.io/egress-bandwidth ，bandwidth 仍然会使用 cni 配置文件的进行限制，这是什么原理呢？

  1. 通过对 bandwidth 源码添加一些日志信息编译后，重新被kubelet 调用后可以看到，传给 bandwidth 二进制文件的配置信息（通过 stdin 传入）中有两个带宽限制的信息，一个是 runtimeConfig 中的，即在 pod.annotation 中指定的 800M；另外一个是配置文件中配置的 300M；而最终采用了 300M 的配置。
  2. bandwidth 源码中的选取原理为：只要配置文件不为空即采用配置文件的；若配置文件为空，即采用 runtimeConfig 的。源码： https://github.com/containernetworking/plugins/blob/main/plugins/meta/bandwidth/main.go#L98-L103

func writeMessage(str string) {
  filePath := "/root/cni.log"
  file, err := os.OpenFile(filePath, os.O_WRONLY|os.O_CREATE|os.O_APPEND, 0666)
  if err != nil {
    _ = fmt.Errorf("文件打开失败, %s", err.Error())
  }
  defer file.Close()

  write := bufio.NewWriter(file)
  write.WriteString(str)
  write.WriteString("\n\n")
  write.Flush()
}

// parseConfig parses the supplied configuration (and prevResult) from stdin.
func parseConfig(stdin []byte) (*PluginConf, error) {
  conf := PluginConf{}
  if err := json.Unmarshal(stdin, &conf); err != nil {
    return nil, fmt.Errorf("failed to parse network configuration: %v", err)
  }

  bs, _ := json.Marshal(conf)
  var out bytes.Buffer
  json.Indent(&out, bs, "", "\t")
  writeMessage(out.String())

  bandwidth := getBandwidth(&conf)
  if bandwidth != nil {
    writeMessage(bandwidth.String())

    err := validateRateAndBurst(bandwidth.IngressRate, bandwidth.IngressBurst)
    if err != nil {
      return nil, err
    }
    err = validateRateAndBurst(bandwidth.EgressRate, bandwidth.EgressBurst)
    if err != nil {
      return nil, err
    }
  }

  if conf.RawPrevResult != nil {
    var err error
    if err = version.ParsePrevResult(&conf.NetConf); err != nil {
      return nil, fmt.Errorf("could not parse prevResult: %v", err)
    }

    _, err = current.NewResultFromResult(conf.PrevResult)
    if err != nil {
      return nil, fmt.Errorf("could not convert result to current version: %v", err)
    }
  }

  return &conf, nil
}
