# runit 服务文件路径：/lib/systemd/system/runit.service
# killMode 修改为默认的 control-group；
# 从文件中加载环境变量 EnvironmentFile

Description=Runit service supervision

EnvironmentFile=/etc/calico/calico.env

    #!/bin/bash

# 安装软件包
apt-get install runit runit-systemd ipset -y

# 修改 runit 服务
cat > /lib/systemd/system/runit.service << EOF
[Unit]
Description=Runit service supervision
Documentation=http://smarden.org/runit/

[Service]
EnvironmentFile=/etc/calico/calico.env
ExecStart=/etc/runit/2
KillSignal=SIGHUP
KillMode=control-group
Restart=always

[Install]
WantedBy=multi-user.target
EOF

# 修改 /etc/runit/2 - 添加注释 #
sed -i '/exec env/s/^/#/' /etc/runit/2


# 拷贝calico-node 容器中的数据到主机中
# 这里也可以从基础镜像中获取；但是环境变量就需要自己去写
mkdir /etc/calico
id=$(docker ps |grep k8s_calico-node | awk '{print $1}')
docker cp $id:/usr/bin/calico-node /usr/local/bin/
docker cp $id:/usr/bin/bird /usr/local/bin/
docker cp $id:/usr/bin/bird6 /usr/local/bin/
docker cp $id:/etc/service/enabled/ /etc/service/
docker cp $id:/etc/calico/ /etc/calico
docker cp $id:/etc/envvars /etc/calico/calico.env

# 修改环境变量
sed -i '/PWD=\//d' /etc/calico/calico.env
sed -i '/HOME=\/root/d' /etc/calico/calico.env
echo KUBECONFIG=/root/.kube/config  >> /etc/calico/calico.env

# 修改 calico-node 使其无法被调度到任何节点 - 随意加 label 或者删除掉也可以
# kubectl delete ds -n -n kube-system calico-node
kubectl patch daemonset -n kube-system calico-node -p '{"spec":{"template":{"spec":{"nodeSelector":{"kubernetes.io/os":"linux","test":"value"}}}}}'

# 修改 cni 配置中的 kubeconfig
sed -i 's|"kubeconfig": "/etc/cni/net.d/calico-kubeconfig"|"kubeconfig": "/root/.kube/config"|g' /etc/cni/net.d/10-calico.conflist
