      cmd: bash  
  nodes:  
    router0:  
      kind: linux  
      image: frrouting/frr:v8.2.2  
      labels:  
        app: frr  
      exec:  
      - iptables -t nat -A POSTROUTING -o eth0 -j MASQUERADE  
      - ip addr add 10.0.0.0/32 dev lo  
      - ip route add blackhole 10.0.0.0/8  
      - touch /etc/frr/vtysh.conf  
      - sed -i -e 's/bgpd=no/bgpd=yes/g' /etc/frr/daemons  
      - usr/lib/frr/frrinit.sh start  
      - >-  
         vtysh -c 'conf t'  
         -c 'router bgp 65000'  
         -c ' bgp router-id 10.0.0.0'  
         -c ' no bgp ebgp-requires-policy'  
         -c ' neighbor ROUTERS peer-group'  
         -c ' neighbor ROUTERS remote-as external'  
         -c ' neighbor ROUTERS default-originate'  
         -c ' neighbor net0 interface peer-group ROUTERS'  
         -c ' neighbor net1 interface peer-group ROUTERS'  
         -c ' address-family ipv4 unicast'  
         -c '   redistribute connected'

         -c '!'

set interfaces ethernet eth0 address '172.20.20.5/24'
set interfaces ethernet eth0 description public
set protocols static route 0.0.0.0/0 next-hop 172.20.20.1 distance 1
set nat source rule 100 outbound-interface 'eth0'
set nat source rule 100 source address '10.1.0.0/16'
set nat source rule 100 translation address 'masquerade'

set interfaces ethernet eth1 address '10.1.10.2/24'
set interfaces ethernet eth2 address '10.1.34.2/24'
set protocols bgp local-as 500 
set protocols bgp neighbor 10.1.10.1 remote-as '65005'
set protocols bgp neighbor 10.1.34.1 remote-as '65008'
set protocols bgp parameters router-id '10.1.10.2'
